$(document).ready(function() {
    var topMain = $("#header").height(); //是头部的高度加头部与nav导航之间的距离
    var nav = $(".nav");
    var type = 0;
    var isShow = 1;
    var isShow2 = 1;
    var prevTop = 0,
    currTop = 0;
    $(window).scroll(function() {
      //如果滚动条顶部的距离大于topMain则就nav导航就添加类.nav_scroll，否则就移除
        if ($(window).scrollTop() > topMain) { 
            //添加
            if (nav.attr('class') == "nav") {
                add();
                nav.attr('style', "background-color:#fff");
            } else {
                currTop = $(window).scrollTop();
                if (currTop < prevTop) { //判断小于则为向上滚动
                    if (nav.is(":hidden")) {
 
                        if (isShow == 1) {
                            isShow = 0;
                            //当前是hide状态
                            nav.animate({
                                height: 'toggle'
                            },
                            {
                                duration: 500,
                                queue: false,
                                complete: function() {
                                    isShow = 1;
                                }
                            });
                        }
                    } else {
                        //当前是show状态
                    }
                } else {
                    if (nav.is(":hidden")) {
                        //当前是hide状态
                    } else {
                        nav.toggle();
                        //当前是show状态
                    }
                }
                //prevTop = currTop; //IE下有BUG，所以用以下方式
                setTimeout(function() {prevTop = currTop},0);
            }
        } else {
            //移除
            if (nav.attr('class') == "nav nav_scroll") {
                remove();
                nav.attr('style', "background-color:rgba(255, 255, 255, 0)");
            }
        }
    });
 
    function add() {
        nav.addClass("nav_scroll");
        type = 0;
    }
    function remove() {
        nav.removeClass("nav_scroll");
        type = 0;
    }
});
