//监听屏幕滚动到底部TOP的显示隐藏
 $(window).scroll(function() {
    if($(this).scrollTop() != 0) {
         $('#top').fadeIn();
         } else {
             $('#top').fadeOut();
         }
    });
 //点击top滚回顶部
 $('#top').click(function() {
    $('body,html').animate({scrollTop:0},500);
     });
